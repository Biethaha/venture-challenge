    function App() {

      const [screen, setScreen] = useState('start');

      const [session, setSession] = useState(null);

      const [username, setUsername] = useState(null);

      const [sessions, setSessions] = useState([]);

      const [savedLogin, setSavedLogin] = useState(null);

 

      useEffect(() => { setSessions(Storage.getAllSessions()); setSavedLogin(Storage.getLogin()); const params = new URLSearchParams(window.location.search); if (params.get('join')) setScreen('student-register'); }, []);

 

      const handleSelectRole = (role, sessionData, user) => {

        if (role === 'admin-new') setScreen('admin-setup');

        else if (role === 'admin' && sessionData) { setSession(sessionData); setScreen('admin'); }

        else if (role === 'student-register') setScreen('student-register');

        else if (role === 'student-login' && sessionData && user) { setSession(sessionData); setUsername(user); setScreen('student'); }

      };

 

      const handleLogout = () => { Storage.clearLogin(); setSession(null); setUsername(null); setScreen('start'); setSessions(Storage.getAllSessions()); setSavedLogin(null); window.history.replaceState({}, '', window.location.pathname); };

 

      return (

        <div style={{ color: '#fff' }}>

          {screen === 'start' && <StartScreen onSelectRole={handleSelectRole} sessions={sessions} savedLogin={savedLogin} />}

          {screen === 'admin-setup' && <AdminSetup onComplete={(s) => { setSession(s); setScreen('admin'); }} onBack={() => setScreen('start')} />}

          {screen === 'student-register' && <StudentRegistration onComplete={(s, u) => { setSession(s); setUsername(u); setScreen('student'); }} onBack={() => setScreen('start')} />}

          {screen === 'admin' && session && <AdminDashboard initialSession={session} onLogout={handleLogout} />}

          {screen === 'student' && session && username && <StudentDashboard initialSession={session} username={username} onLogout={handleLogout} />}

        </div>

      );

    }

 

    ReactDOM.createRoot(document.getElementById('root')).render(<App />);

  </script>
